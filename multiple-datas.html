多組資料

多組資料時，寫成雜湊物件的樣式，如下：

var data_1 = [
  [1325347200000, 60], [1328025600000, 100], [1330531200000, 15], [1333209600000, 50]
];
 
var data_2 = [
   [1325347200000, 5], [1328025600000, 12], [1330531200000, 7], [1333209600000, 25]
 ];
 
var dataset = [
  {
     label: "資料1",
     data: data_1
  },
  {
     label: "資料2",
     data: data_2
  },
];
 
var options = {
  xaxis: {
    mode: "time"
 }
};
 
$(function() {
  $.plot($("#chart"), dataset, options);
});
