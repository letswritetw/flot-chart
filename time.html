時間折線圖

  1. 時間格式為timestamps格式。
  2. axes裡的mode必須設定為time。
  3. 需引用jquery.flot.time.js。

timestamps用法

// 取得從1970/1/1 00:00:00開始計算的毫秒總量(時間戳記)
var d = new Date(2012,1,2); // 裡面可以寫時間，沒有寫的話就是「現在」
var time = d.getTime();

引用

<script src="jquery.min.js"></script>
<script src="jquery.flot.min.js"></script>
<script src="jquery.flot.time.js"></script>
 
// 如果要有動畫，就要引用以下
<script src="jquery.flot.animator.min.js"></script>
 
<!--[if lt IE 8]>
	<script src="excanvas.js"></script>
<![endif]-->

使用範例：

var data = [
  [1325347200000, 60], [1328025600000, 100], [1330531200000, 15], [1333209600000, 50]
];
 
var dataset = [{
  label: "時間數據",
  data: data
}];
 
var options = {
  xaxis: {
    mode: "time"
 }
};
 
$(function() {
  $.plot($("#chart"), dataset, options);
});
