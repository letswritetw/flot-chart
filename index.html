基本使用 - flot demo

引用 ( 檔案下載 )

<script src="jquery.min.js"></script>
<script src="jquery.flot.min.js"></script>
 
// 如果要有動畫，就要引用以下
<script src="jquery.flot.animator.min.js"></script>
 
<!--[if lt IE 8]>
	<script src="excanvas.js"></script>
<![endif]-->

使用

先用一個div帶上id:

<div class="charts-block">
  <div id="chart"> <div>
<div>

建議是在外層包一個div，並設定寬、高，就像上面的.charts-block

使用的js如下：

/*
   target：要顯示的位置，在這裡就是#chart。
   data：顯示的數據。
   options：其它選項。
*/
$.plot('target', data, options);
 
// 如果要有動畫，就改成以下
$.plotAnimator('target', data, options);

使用範例：

var data = [
  /*
    資料格式是由多個資料序列組成的陣列：[series1, series2, series3, ...]
    而每個序列的資料是由x軸和y軸的數值組合而成：[X, Y]
  */
  [1, 130], [2, 40], [3, 80], [4, 160], [5, 159], [6, 370], [7, 330]
];
 
var dataset = [{
  label: "數據",
  data: data
 
  // 如要設定動畫，option如下
  // animator: { start: 1, steps: 99, duration: 2000, direction: "right" },
}];
 
var options = {
  series: {
    lines: {
      show: true
    },
    points: {
      radius: 3,
      show: true
    }
  }
};
 
$(function() {
  $.plotAnimator($("#chart"), dataset, options);
});
